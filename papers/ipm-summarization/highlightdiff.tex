\begin{abstract}
\begin{itemize}
\item \DIFdelbegin \DIFdel{The }\DIFdelend \DIFaddbegin \DIFadd{We identify the }\DIFaddend repetition problem in abstractive \DIFdelbegin \DIFdel{summaries generated by }\DIFdelend \DIFaddbegin \DIFadd{summarization based on }\DIFaddend CNN seq2seq models.
\item \DIFdelbegin \DIFdel{New }\DIFdelend \DIFaddbegin \DIFadd{We propose two new }\DIFaddend evaluation metrics: Repeatedness and Readability. 
\item \DIFdelbegin \DIFdel{Repetition Reduction: The }\DIFdelend \DIFaddbegin \DIFadd{We propose an effective repetition reduction approach that consists of the }\DIFaddend Attention Filter mechanism (ATTF) \DIFdelbegin \DIFdel{at training }\DIFdelend and Sentence-level Backtracking Decoder (SBD)\DIFdelbegin \DIFdel{at runtime}\DIFdelend .
\item \DIFdelbegin \DIFdel{The improvement of the quality of abstractive summaries after reducing repetition by ATTF and SBD}\DIFdelend \DIFaddbegin \DIFadd{Our proposed approach outperforms the state-of-the-art (SOTA) CNN-based baselines substantially by all evaluation metrics}\DIFaddend .
\end{itemize}
\end{abstract}
