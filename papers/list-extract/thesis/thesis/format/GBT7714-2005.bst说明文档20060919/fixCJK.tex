%This file fixes problems with Bibtex and CJK in displaying CJK characters
%from CJK's author: Werner Lemberg <wl@gnu.org>
%The following codes will be available in future versions of CJK
\makeatletter
\AtBeginDocument{%
  \if@filesw
    \immediate\write\@mainaux{%
      \string\begin{CJK}{GBK}{song}}%
    \immediate\write\@mainaux{%
      \string\makeatletter}%
  \fi}
\AtEndDocument{%
  \if@filesw
    \immediate\write\@mainaux{%
      \string\end{CJK}}%
  \fi}
\makeatother

\endinput
